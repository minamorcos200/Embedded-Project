void delay_ms(int n)
{
    int i,j;
    for(i=0;i<n;i++)
        for(j=0;j<3180;j++)
        {}
}


void delay_us(int n)
{
    int i,j;
    for(i=0;i<n;i++)
        for(j=0;j<3;j++)
        {}
}

void LCD_command(unsigned char com)
{

    GPIO_PORTA_DATA_R=0;        
    delay_ms(1);
    GPIO_PORTA_DATA_R |=0x40;   
    delay_ms(1);
    GPIO_PORTB_DATA_R=com;     
    delay_us(1);
    GPIO_PORTA_DATA_R=0;       
    delay_us(1);

}


void LCD_data(unsigned char data)
{
    GPIO_PORTA_DATA_R=0x80;        
    delay_ms(1);
    GPIO_PORTA_DATA_R |=0x40;      
    delay_ms(1);
    GPIO_PORTB_DATA_R=data;        
    delay_us(1);
    GPIO_PORTA_DATA_R=0;           
    delay_us(50);
}

void LCD_string(char *str) 
{
    int i;
    for(i=0;str[i]!=0;i++)  
    {
        LCD_data(str[i]);  
        delay_ms(20);
    }
}

void LCD_int(int data)
{
    int p;
    int k=0;
    while(data>0)
    {
        num[k]=data%10;
        data=data/10;
        k++;
    }
    k--;
    for (p=k;p>=0;p--)
    {
        c=num[p]+48;

        GPIO_PORTA_DATA_R=0x80;       
        delay_ms(1);
        GPIO_PORTA_DATA_R |=0x40;      
        delay_ms(1);
        GPIO_PORTB_DATA_R=c;           
        delay_us(1);
        GPIO_PORTA_DATA_R=0;       
        delay_us(50);
    }
}


void LCD_separating_double(double deci)
{
    int integer_part=deci;
    int decimal_part=(deci - integer_part)*100000;

    LCD_int(integer_part);
    LCD_data('.');
    LCD_int(decimal_part);
}



float deg_to_rad(float deg){
    return (deg * PI / 180);
}


float total_Dist(float long1, float long2, float lat1, float lat2)
{
    float dlong = deg_to_rad(long2 - long1);
    float dlat  = deg_to_rad(lat2 - lat1);
    float phi1 = deg_to_rad(lat1);
    float phi2 = deg_to_rad(lat2);
   
    float a = pow(sin((0.5 * dlat)), 2) + cos(phi1) * cos(phi2) * pow(sin((0.5 * dlong)), 2);
    float d = 2 * R * asin(sqrt(a));
    Dis = Dis + d;
    return Dis;
}
